│ GoChannel 实现设计                                                           │
│                                                                              │
│ // internal/queue/channel/queue.go                                           │
│ package channel                                                              │
│                                                                              │
│ import (                                                                     │
│     "context"                                                                │
│     "sync"                                                                   │
│     "time"                                                                   │
│                                                                              │
│     "github.com/Done-0/gin-scaffold/internal/queue"                          │

│                                                                              │
│ // Broker manages in-memory message channels                                 │
│ type Broker struct {                                                         │
│     mu       sync.RWMutex                                                    │
│     channels map[string]chan *queue.Message                                  │
│     bufSize  int                                                             │
│ }                                                                            │
│                                                                              │
│ // Producer sends messages to in-memory channels                             │
│ type Producer struct {                                                       │
│     broker *Broker                                                           │
│ }                                                                            │
│                                                                              │
│ // Consumer receives messages from in-memory channels                        │
│ type Consumer struct {                                                       │
│     broker  *Broker                                                          │
│     handler queue.Handler                                                    │
│     ctx     context.Context                                                  │
│     cancel  context.CancelFunc                                               │
│ }                                                                            │
│                                                                              │
│ var globalBroker *Broker                                                     │
│ var once sync.Once                                                           │
│                                                                              │
│ func GetBroker() *Broker {                                                   │
│     once.Do(func() {                                                         │
│         globalBroker = &Broker{                                              │
│             channels: make(map[string]chan *queue.Message),                  │
│             bufSize:  100, // 缓冲区大小                                     │
│         }                                                                    │
│     })                                                                       │
│     return globalBroker                                                      │
│ }                                                                            │

│ - 使用 routing key = queue name                                              │
